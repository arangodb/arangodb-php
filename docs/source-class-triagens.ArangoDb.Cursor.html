168     /**
169      * Initialise the cursor with the first results and some metadata
170      *
171      * @param Connection $connection - connection to be used
172      * @param array $data - initial result data as returned by the server
173      * @param array $options - cursor options
174      *
175      * @return Cursor
176      */
177     public function __construct(Connection $connection, array $data, array $options)
178     {
179         $this->_connection = $connection;
180         $this->data        = $data;
181         $this->_id         = null;
182         $this->_extra      = array();
183         $this->_cached     = false;
184 
185         if (isset($data[self::ENTRY_ID])) {
186             $this->_id = $data[self::ENTRY_ID];
187         }
188 
189         if (isset($data[self::ENTRY_EXTRA])) {
190             // ArangoDB 2.3+ return value struct
191             $this->_extra = $data[self::ENTRY_EXTRA];
192 
193             if (isset($this->_extra[self::ENTRY_STATS][self::FULL_COUNT])) {
194                 $this->_fullCount = $this->_extra[self::ENTRY_STATS][self::FULL_COUNT];
195             }
196         }
197         else if (isset($data[self::ENTRY_EXTRA][self::FULL_COUNT])) {
198             // pre-ArangoDB 2.3 return value struct
199             $this->_fullCount = $data[self::ENTRY_EXTRA][self::FULL_COUNT];
200         }
201 
202         if (isset($data[self::ENTRY_CACHED])) {
203             $this->_cached = $data[self::ENTRY_CACHED];
204         }
205 
206         // attribute must be there
207         assert(isset($data[self::ENTRY_HASMORE]));
208         $this->_hasMore = (bool) $data[self::ENTRY_HASMORE];
209 
210         $options['isNew'] = false;
211         $this->_options   = $options;
212         $this->_result    = array();
213         $this->add((array) $data[self::ENTRY_RESULT]);
214         $this->updateLength();
215 
216         $this->rewind();
217     }

346     /**
347      * Check if cursor can be advanced further, necessary for Iterator
348      *
349      * This might issue additional HTTP requests to fetch any outstanding
350      * results from the server
351      *
352      * @throws Exception
353      * @return bool - true if the cursor can be advanced further, false if cursor is at end
354      */
355     public function valid()
356     {
357         if ($this->_position <= $this->_length - 1) {
358             // we have more results than the current position is
359             return true;
360         }
361 
362         if (!$this->_hasMore || !$this->_id) {
363             // we don't have more results, but the cursor is exhausted
364             return false;
365         }
366 
367         // need to fetch additional results from the server
368         $this->fetchOutstanding();
369 
370         return ($this->_position <= $this->_length - 1);
371     }

374     /**
375      * Create an array of results from the input array
376      *
377      * @param array $data - incoming result
378      *
379      * @return void
380      */
381     private function add(array $data)
382     {
383         foreach ($this->sanitize($data) as $row) {
384             if (!is_array($row) || (isset($this->_options[self::ENTRY_FLAT]) && $this->_options[self::ENTRY_FLAT])) {
385                 $this->addFlatFromArray($row);
386             }
387             else {
388                 if (!isset($this->_options['objectType'])) {
389                     $this->addDocumentsFromArray($row);
390                 }
391                 else {
392                     switch ($this->_options['objectType']) {
393                         case 'edge' :
394                             $this->addEdgesFromArray($row);
395                             break;
396                         case 'vertex' :
397                             $this->addVerticesFromArray($row);
398                             break;
399                         case 'path' :
400                             $this->addPathsFromArray($row);
401                             break;
402                         case 'shortestPath' :
403                             $this->addShortestPathFromArray($row);
404                             break;
405                         case 'distanceTo' :
406                             $this->addDistanceToFromArray($row);
407                             break;
408                         case 'commonNeighbors' :
409                             $this->addCommonNeighborsFromArray($row);
410                             break;
411                         case 'commonProperties' :
412                             $this->addCommonPropertiesFromArray($row);
413                             break;
414                         case 'figure' :
415                             $this->addFigureFromArray($row);
416                             break;
417                         default :
418                             $this->addDocumentsFromArray($row);
419                             break;
420                     }
421                 }
422             }
423         }
424     }

459     private function addPathsFromArray(array $data)
460     {
461         $entry = array(

463             'edges' => array(),
464             'source' => Document::createFromArray($data['source'], $this->_options),
465             'destination' => Document::createFromArray($data['destination'], $this->_options),
466         );
467         foreach ($data['vertices'] as $v) {
468             $entry['vertices'][] = Document::createFromArray($v, $this->_options);
469         }
470         foreach ($data['edges'] as $v) {
471             $entry['edges'][] = Edge::createFromArray($v, $this->_options);
472         }
473         $this->_result[] = $entry;

476     /**
477      * Create an array of shortest paths from the input array
478      *
479      * @param array $data - array of incoming "paths" arrays
480      *
481      * @return void
482      */
483     private function addShortestPathFromArray(array $data)
484     {
485         if (!isset($data['vertices'])) {
486             return;
487         }
488 
489         $vertices    = $data['vertices'];
490         $startVertex = $vertices[0];
491         $destination = $vertices[count($vertices) - 1];
492 
493         $entry = array(
494             'paths' => array(),
495             'source' => Document::createFromArray($startVertex, $this->_options),
496             'distance' => $data['distance'],
497             'destination' => Document::createFromArray($destination, $this->_options),
498         );
499 
500         $path = array(
501             'vertices' => array(),
502             'edges' => array()
503         );
504 
505         foreach ($data['vertices'] as $v) {
506             $path['vertices'][] = $v;
507         }
508         foreach ($data['edges'] as $v) {
509             $path['edges'][] = Edge::createFromArray($v, $this->_options);
510         }
511         $entry['paths'][] = $path;
512 
513         $this->_result[] = $entry;
514     }

534     /**
535      * Create an array of common neighbors from the input array
536      *
537      * @param array $data - array of incoming "paths" arrays
538      *
539      * @return void
540      */
541     private function addCommonNeighborsFromArray(array $data)
542     {
543         $left  = $data['left'];
544         $right = $data['right'];
545 
546         if (!isset($this->_result[$left])) {
547             $this->_result[$left] = array();
548         }
549         if (!isset($this->_result[$left][$right])) {
550             $this->_result[$left][$right] = array();
551         }
552 
553         foreach ($data['neighbors'] as $neighbor) {
554             $this->_result[$left][$right][] = Document::createFromArray($neighbor);
555         }
556     }

615     /**
616      * Sanitize the result set rows
617      *
618      * This will remove the _id and _rev attributes from the results if the
619      * "sanitize" option is set
620      *
621      * @param array $rows - array of rows to be sanitized
622      *
623      * @return array - sanitized rows
624      */
625     private function sanitize(array $rows)
626     {
627         if (isset($this->_options[self::ENTRY_SANITIZE]) && $this->_options[self::ENTRY_SANITIZE]) {
628             foreach ($rows as $key => $value) {
629 
630                 if (is_array($value) && isset($value[Document::ENTRY_ID])) {
631                     unset($rows[$key][Document::ENTRY_ID]);
632                 }
633 
634                 if (is_array($value) && isset($value[Document::ENTRY_REV])) {
635                     unset($rows[$key][Document::ENTRY_REV]);
636                 }
637             }
638         }
639 
640         return $rows;
641     }

644     /**
645      * Fetch outstanding results from the server
646      *
647      * @throws Exception
648      * @return void
649      */
650     private function fetchOutstanding()
651     {
652         // continuation
653         $response = $this->_connection->put($this->url() . '/' . $this->_id, '', array());
654         ++$this->_fetches;
655 
656         $data = $response->getJson();
657 
658         $this->_hasMore = (bool) $data[self::ENTRY_HASMORE];
659         $this->add($data[self::ENTRY_RESULT]);
660 
661         if (!$this->_hasMore) {
662             // we have fetched the complete result set and can unset the id now
663             $this->_id = null;
664         }
665 
666         $this->updateLength();
667     }
