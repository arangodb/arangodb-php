  88     /**
  89      * Create a graph
  90      *
  91      * This will create a graph using the given graph object and return an array of the created graph object's attributes.<br><br>
  92      *
  93      * @throws Exception
  94      *
  95      * @param Graph $graph - The graph object which holds the information of the graph to be created
  96      *
  97      * @return array
  98      * @since   1.2
  99      */
 100     public function createGraph(Graph $graph)
 101     {
 102         $edgeDefinitions = array();
 103         foreach ($graph->getEdgeDefinitions() as $ed) {
 104             $edgeDefinitions[] = $ed->transformToArray();
 105         }
 106 
 107         $params   = array(
 108             self::OPTION_NAME => $graph->getKey(),
 109             self::OPTION_EDGE_DEFINITIONS => $edgeDefinitions,
 110             self::OPTION_ORPHAN_COLLECTIONS => $graph->getOrphanCollections()
 111         );
 112         $url      = Urls::URL_GRAPH;
 113         $response = $this->getConnection()->post($url, $this->json_encode_wrapper($params));
 114         $json     = $response->getJson();
 115         $graph->setInternalId($json['graph'][Graph::ENTRY_ID]);
 116         $graph->set(Graph::ENTRY_KEY, $json['graph'][self::OPTION_NAME]);
 117         $graph->setRevision($json['graph'][Graph::ENTRY_REV]);
 118 
 119 
 120         return $graph->getAll();
 121     }

 530     /**
 531      * save a vertex to a graph
 532      *
 533      * This will add the vertex-document to the graph and return the vertex id
 534      *
 535      * This will throw if the vertex cannot be saved<br><br>
 536      *
 537      * @throws Exception
 538      *
 539      * @param mixed $graph - graph name as a string or instance of Graph
 540      * @param mixed $document - the vertex to be added, can be passed as a vertex object or an array
 541      * @param string $collection - if one uses a graph with more than one vertex collection one must provide
 542      * the collection to store the vertex.
 543      *
 544      * @return string - id of vertex created
 545      * @since 1.2
 546      */
 547     public function saveVertex($graph, $document, $collection = null)
 548     {
 549         if ($graph instanceof Graph) {
 550             $graph = $graph->getKey();
 551         }
 552 
 553         if (is_array($document)) {
 554             $document = Vertex::createFromArray($document);
 555         }
 556         if ($collection === null) {
 557             if (count($this->getVertexCollections($graph)) !== 1) {
 558                 throw new ClientException('A collection must be provided.');
 559             }
 560             else if (count($this->getVertexCollections($graph)) === 1) {
 561                 $collection = $this->getVertexCollections($graph);
 562                 $collection = $collection[0];
 563             }
 564         }
 565         $data = $document->getAll();
 566         $url  = UrlHelper::buildUrl(Urls::URL_GRAPH, array($graph, Urls::URLPART_VERTEX, $collection));
 567 
 568         $response = $this->getConnection()->post($url, $this->json_encode_wrapper($data));
 569 
 570         $jsonArray = $response->getJson();
 571         $vertex    = $jsonArray['vertex'];
 572 
 573         $document->setInternalId($vertex[Vertex::ENTRY_ID]);
 574         $document->setRevision($vertex[Vertex::ENTRY_REV]);
 575 
 576         $document->setIsNew(false);
 577 
 578         return $document->getInternalId();
 579     }

 582     /**
 583      * Get a single vertex from a graph
 584      *
 585      * This will throw if the vertex cannot be fetched from the server<br><br>
 586      *
 587      * @throws Exception
 588      *
 589      * @param mixed $graph - graph name as a string or instance of Graph
 590      * @param mixed $vertexId - the vertex identifier
 591      * @param array $options optional, an array of options:
 592      * <p>
 593      * <li><b>_includeInternals</b> - true to include the internal attributes. Defaults to false</li>
 594      * <li><b>includeInternals</b> - Deprecated, please use '_includeInternals'.</li>
 595      * <li><b>_ignoreHiddenAttributes</b> - true to show hidden attributes. Defaults to false</li>
 596      * <li><b>ignoreHiddenAttributes</b> - Deprecated, please use '_ignoreHiddenAttributes'.</li>
 597      * </p>
 598      * @param string $collection - if one uses a graph with more than one vertex collection one must provide the collection
 599      *  to load the vertex.
 600      *
 601      * @return Document
 602      * @since 1.2
 603      */
 604     public function getVertex($graph, $vertexId, array $options = array(), $collection = null)
 605     {
 606         if ($graph instanceof Graph) {
 607             $graph = $graph->getKey();
 608         }
 609         $parts = explode('/', $vertexId);
 610         if (count($parts) === 2) {
 611             $vertexId   = $parts[1];
 612             $collection = $parts[0];
 613         }
 614         if ($collection === null) {
 615             if (count($this->getVertexCollections($graph)) !== 1) {
 616                 throw new ClientException('A collection must be provided.');
 617             }
 618             else if (count($this->getVertexCollections($graph)) === 1) {
 619                 $collection = $this->getVertexCollections($graph);
 620                 $collection = $collection[0];
 621             }
 622         }
 623         $url      = UrlHelper::buildUrl(Urls::URL_GRAPH, array($graph, Urls::URLPART_VERTEX, $collection, $vertexId));
 624         $response = $this->getConnection()->get($url);
 625 
 626         $jsonArray = $response->getJson();
 627         $vertex    = $jsonArray['vertex'];
 628 
 629         $options['_isNew'] = false;
 630 
 631         return Vertex::createFromArray($vertex, $options);
 632     }
