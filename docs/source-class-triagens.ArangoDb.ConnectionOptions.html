356     /**
357      * Validate the options
358      *
359      * @throws ClientException
360      * @return void - will throw if an invalid option value is found
361      */
362     private function validate()
363     {
364         if (isset($this->_values[self::OPTION_HOST]) && !is_string($this->_values[self::OPTION_HOST])) {
365             throw new ClientException('host should be a string');
366         }
367 
368         if (isset($this->_values[self::OPTION_PORT]) && !is_int($this->_values[self::OPTION_PORT])) {
369             throw new ClientException('port should be an integer');
370         }
371 
372         // can use either endpoint or host/port
373         if (isset($this->_values[self::OPTION_HOST]) && isset($this->_values[self::OPTION_ENDPOINT])) {
374             throw new ClientException('must not specify both host and endpoint');
375         }
376         else {
377             if (isset($this->_values[self::OPTION_HOST]) && !isset($this->_values[self::OPTION_ENDPOINT])) {
378                 // upgrade host/port to an endpoint
379                 $this->_values[self::OPTION_ENDPOINT] = 'tcp://' . $this->_values[self::OPTION_HOST] . ':' . $this->_values[self::OPTION_PORT];
380             }
381         }
382 
383         assert(isset($this->_values[self::OPTION_ENDPOINT]));
384         // set up a new endpoint, this will also validate it
385         $this->getEndpoint();
386 
387         $type = Endpoint::getType($this->_values[self::OPTION_ENDPOINT]);
388         if ($type === Endpoint::TYPE_UNIX) {
389             // must set port to 0 for UNIX domain sockets
390             $this->_values[self::OPTION_PORT] = 0;
391         }
392         elseif ($type === Endpoint::TYPE_SSL) {
393             // must set port to 0 for SSL connections
394             $this->_values[self::OPTION_PORT] = 0;
395         }
396 
397         if (isset($this->_values[self::OPTION_AUTH_TYPE]) && !in_array(
398                 $this->_values[self::OPTION_AUTH_TYPE],
399                 self::getSupportedAuthTypes()
400             )
401         ) {
402             throw new ClientException('unsupported authorization method');
403         }
404 
405         if (isset($this->_values[self::OPTION_CONNECTION]) && !in_array(
406                 $this->_values[self::OPTION_CONNECTION],
407                 self::getSupportedConnectionTypes()
408             )
409         ) {
410             throw new ClientException(
411                 sprintf(
412                     "unsupported connection value '%s'",
413                     $this->_values[self::OPTION_CONNECTION]
414                 )
415             );
416         }
417 
418         UpdatePolicy::validate($this->_values[self::OPTION_UPDATE_POLICY]);
419         UpdatePolicy::validate($this->_values[self::OPTION_REPLACE_POLICY]);
420         UpdatePolicy::validate($this->_values[self::OPTION_DELETE_POLICY]);
421     }
