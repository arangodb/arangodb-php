 73     /**
 74      * Create a one-time HTTP connection by opening a socket to the server
 75      *
 76      * It is the caller's responsibility to close the socket
 77      *
 78      * @throws ConnectException
 79      *
 80      * @param ConnectionOptions $options - connection options
 81      *
 82      * @return resource - socket with server connection, will throw when no connection can be established
 83      */
 84     public static function createConnection(ConnectionOptions $options)
 85     {
 86         $endpoint = $options[ConnectionOptions::OPTION_ENDPOINT];
 87 
 88         $context = stream_context_create();
 89 
 90         if (Endpoint::getType($endpoint) === Endpoint::TYPE_SSL) {
 91             // set further SSL options for the endpoint
 92             stream_context_set_option($context, 'ssl', 'verify_peer', $options[ConnectionOptions::OPTION_VERIFY_CERT]);
 93             stream_context_set_option($context, 'ssl', 'allow_self_signed', $options[ConnectionOptions::OPTION_ALLOW_SELF_SIGNED]);
 94 
 95             if ($options[ConnectionOptions::OPTION_CIPHERS] !== null) {
 96                 // SSL ciphers
 97                 stream_context_set_option($context, 'ssl', 'ciphers', $options[ConnectionOptions::OPTION_CIPHERS]);
 98             }
 99         }
100 
101         $fp = @stream_socket_client(
102             $endpoint,
103             $errno,
104             $message,
105             $options[ConnectionOptions::OPTION_TIMEOUT],
106             STREAM_CLIENT_CONNECT,
107             $context
108         );
109 
110         if (!$fp) {
111             throw new ConnectException(
112                 'cannot connect to endpoint \'' .
113                 $options[ConnectionOptions::OPTION_ENDPOINT] . '\': ' . $message, $errno
114             );
115         }
116 
117         stream_set_timeout($fp, $options[ConnectionOptions::OPTION_TIMEOUT]);
118 
119         return $fp;
120     }

132     /**
133      * Create a request string (header and body)
134      *
135      * @param ConnectionOptions $options - connection options
136      * @param string $connectionHeader - preassembled header string for connection
137      * @param string $method - HTTP method
138      * @param string $url - HTTP URL
139      * @param string $body - optional body to post
140      * @param array $customHeaders - any array containing header elements
141      * @return string - assembled HTTP request string
142      * @throws ClientException
143      *
144      */
145     public static function buildRequest(ConnectionOptions $options, $connectionHeader, $method, $url, $body, $customHeaders = array())
146     {
147         if (!is_string($body)) {
148             throw new ClientException('Invalid value for body. Expecting string, got ' . gettype($body));
149         }
150 
151         $length = strlen($body);
152 
153         if ($options[ConnectionOptions::OPTION_BATCH] === true) {
154             $contentType = 'Content-Type: multipart/form-data; boundary=' . self::MIME_BOUNDARY . self::EOL;
155         }
156         else {
157             if ($length > 0 && $options[ConnectionOptions::OPTION_BATCHPART] === false) {
158                 // if body is set, we should set a content-type header
159                 $contentType = 'Content-Type: application/json' . self::EOL;
160             }
161             else {
162                 $contentType = '';
163             }
164         }
165 
166         $customHeader = '';
167         foreach ($customHeaders as $headerKey => $headerValue) {
168             $customHeader .= $headerKey . ': ' . $headerValue . self::EOL;
169         }
170 
171         // finally assemble the request
172         $request = sprintf('%s %s %s', $method, $url, self::PROTOCOL) .
173             $connectionHeader .   // note: this one starts with an EOL
174             $customHeader .
175             $contentType .
176             sprintf('Content-Length: %s', $length) . self::EOL . self::EOL .
177             $body;
178 
179         return $request;
180     }

206     /**
207      * Execute an HTTP request on an opened socket
208      *
209      * It is the caller's responsibility to close the socket
210      *
211      * @param resource $socket - connection socket (must be open)
212      * @param string $request - complete HTTP request as a string
213      *
214      * @throws ClientException
215      * @return string - HTTP response string as provided by the server
216      */
217     public static function transfer($socket, $request)
218     {
219         if (!is_resource($socket)) {
220             throw new ClientException('Invalid socket used');
221         }
222 
223         assert(is_string($request));
224 
225         @fwrite($socket, $request);
226         @fflush($socket);
227 
228         $contentLength    = null;
229         $expectedLength   = null;
230         $totalRead        = 0;
231         $contentLengthPos = 0;
232 
233         $result = '';
234         $first  = true;
235 
236         while ($first || !feof($socket)) {
237             $read = @fread($socket, self::CHUNK_SIZE);
238             if ($read === false || $read === '') {
239                 break;
240             }
241 
242             $totalRead += strlen($read);
243 
244             if ($first) {
245                 $result = $read;
246                 $first  = false;
247             }
248             else {
249                 $result .= $read;
250             }
251 
252             if ($contentLength === null) {
253                 // check if content-length header is present
254 
255                 // 12 = minimum offset (i.e. strlen("HTTP/1.1 xxx") -
256                 // after that we could see "content-length:"
257                 $pos = stripos($result, 'content-length: ', 12);
258 
259                 if ($pos !== false) {
260                     $contentLength    = (int) substr($result, $pos + 16, 10); // 16 = strlen("content-length: ")
261                     $contentLengthPos = $pos + 17; // 17 = 16 + 1 one digit
262                 }
263             }
264 
265             if ($contentLength !== null && $expectedLength === null) {
266                 $bodyStart = strpos($result, "\r\n\r\n", $contentLengthPos);
267                 if ($bodyStart !== false) {
268                     $bodyStart += 4; // 4 = strlen("\r\n\r\n")
269                     $expectedLength = $bodyStart + $contentLength;
270                 }
271             }
272 
273             if ($expectedLength !== null && $totalRead >= $expectedLength) {
274                 break;
275             }
276         }
277 
278         return $result;
279     }

298     /**
299      * Process a string of HTTP headers into an array of header => values.
300      *
301      * @param string $headers - the headers string
302      *
303      * @return array
304      */
305     public static function parseHeaders($headers)
306     {
307         $httpCode  = null;
308         $result    = null;
309         $processed = array();
310 
311         foreach (explode(HttpHelper::EOL, $headers) as $lineNumber => $line) {
312             if ($lineNumber === 0) {
313                 // first line of result is special
314                 if (preg_match("/^HTTP\/\d+\.\d+\s+(\d+)/", $line, $matches)) {
315                     $httpCode = (int) $matches[1];
316                 }
317                 $result = $line;
318             }
319             else {
320                 // other lines contain key:value-like headers
321                 // the following is a performance optimization to get rid of
322                 // the two trims (which are expensive as they are executed over and over) 
323                 if (strpos($line, ': ') !== false) {
324                     list($key, $value) = explode(': ', $line, 2);
325                 }
326                 else {
327                     list($key, $value) = explode(':', $line, 2);
328                 }
329                 $processed[strtolower($key)] = $value;
330             }
331         }
332 
333         return array($httpCode, $result, $processed);
334     }
