 67     /**
 68      * Set a document attribute
 69      *
 70      * The key (attribute name) must be a string.
 71      *
 72      * This will validate the value of the attribute and might throw an
 73      * exception if the value is invalid.
 74      *
 75      * @throws ClientException
 76      *
 77      * @param string $key - attribute name
 78      * @param mixed $value - value for attribute
 79      *
 80      * @return void
 81      */
 82     public function set($key, $value)
 83     {
 84         if ($this->_doValidate) {
 85             // validate the value passed
 86             ValueValidator::validate($value);
 87         }
 88 
 89         if ($key[0] === '_') {
 90             if ($key === self::ENTRY_ID) {
 91                 $this->setInternalId($value);
 92                 return;
 93             }
 94 
 95             if ($key === self::ENTRY_KEY) {
 96                 $this->setInternalKey($value);
 97                 return;
 98             }
 99 
100             if ($key === self::ENTRY_REV) {
101                 $this->setRevision($value);
102                 return;
103             }
104 
105             if ($key === self::ENTRY_FROM) {
106                 $this->setFrom($value);
107                 return;
108             }
109 
110             if ($key === self::ENTRY_TO) {
111                 $this->setTo($value);
112                 return;
113             }
114         }
115 
116         if (!$this->_changed) {
117             if (!isset($this->_values[$key]) || $this->_values[$key] !== $value) {
118                 // set changed flag
119                 $this->_changed = true;
120             }
121         }
122 
123         // and store the value
124         $this->_values[$key] = $value;
125     }
