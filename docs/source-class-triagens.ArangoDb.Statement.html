 13 /**
 14  * Container for an AQL query
 15  *
 16  * Optional bind parameters can be used when issuing thei AQL query to separate
 17  * the query from the values.
 18  * Executing a query will result in a cursor being created.
 19  *
 20  * There is an important distinction between two different types of statements:
 21  * <ul>
 22  * <li> statements that produce an array of documents as their result AND<br />
 23  * <li> statements that do not produce documents
 24  * </ul>
 25  *
 26  * For example, a statement such as "FOR e IN example RETURN e" will produce
 27  * an array of documents as its result. The result can be treated as an array of
 28  * documents, and each document can be updated and sent back to the server by
 29  * the client.<br />
 30  * <br />
 31  * However, the query "RETURN 1 + 1" will not produce an array of documents as
 32  * its result, but an array with a single scalar value (the number 2).
 33  * "2" is not a valid document so creating a document from it will fail.<br />
 34  * <br />
 35  * To turn the results of this query into a document, the following needs to
 36  * be done:
 37  * <ul>
 38  * <li> modify the query to "RETURN { value: 1 + 1 }". The result will then be a
 39  *   an array of documents with a "value" attribute<br />
 40  * <li> use the "_flat" option for the statement to indicate that you don't want
 41  *   to treat the statement result as an array of documents, but as a flat array
 42  * </ul>
 43  *
 44  * <br />
 45  *
 46  * @package triagens\ArangoDb
 47  * @since   0.2
 48  */

160     /**
161      * Initialise the statement
162      *
163      * The $data property can be used to specify the query text and further
164      * options for the query.
165      *
166      * An important consideration when creating a statement is whether the
167      * statement will produce a list of documents as its result or any other
168      * non-document value. When a statement is created, by default it is
169      * assumed that the statement will produce documents. If this is not the
170      * case, executing a statement that returns non-documents will fail.
171      *
172      * To explicitly mark the statement as returning non-documents, the '_flat'
173      * option should be specified in $data.
174      *
175      * @throws Exception
176      *
177      * @param Connection $connection - the connection to be used
178      * @param array $data - statement initialization data
179      */
180     public function __construct(Connection $connection, array $data)
181     {
182         $this->_connection = $connection;
183         $this->_bindVars   = new BindVars();
184 
185         if (isset($data[self::ENTRY_QUERY])) {
186             $this->setQuery($data[self::ENTRY_QUERY]);
187         }
188 
189         if (isset($data[self::ENTRY_COUNT])) {
190             $this->setCount($data[self::ENTRY_COUNT]);
191         }
192 
193         if (isset($data[self::ENTRY_BATCHSIZE])) {
194             $this->setBatchSize($data[self::ENTRY_BATCHSIZE]);
195         }
196 
197         if (isset($data[self::ENTRY_BINDVARS])) {
198             $this->_bindVars->set($data[self::ENTRY_BINDVARS]);
199         }
200 
201         if (isset($data[self::FULL_COUNT])) {
202             $this->_fullCount = (bool) $data[Cursor::FULL_COUNT];
203         }
204 
205         if (isset($data[Cursor::ENTRY_SANITIZE])) {
206             $this->_sanitize = (bool) $data[Cursor::ENTRY_SANITIZE];
207         }
208 
209         if (isset($data[self::ENTRY_RETRIES])) {
210             $this->_retries = (int) $data[self::ENTRY_RETRIES];
211         }
212 
213         if (isset($data[Cursor::ENTRY_FLAT])) {
214             $this->_flat = (bool) $data[Cursor::ENTRY_FLAT];
215         }
216 
217         if (isset($data[Cursor::ENTRY_CACHE])) {
218             $this->_cache = (bool) $data[Cursor::ENTRY_CACHE];
219         }
220     }

232     /**
233      * Execute the statement
234      *
235      * This will post the query to the server and return the results as
236      * a Cursor. The cursor can then be used to iterate the results.
237      *
238      * @throws Exception
239      * @return Cursor
240      */
241     public function execute()
242     {
243         if (!is_string($this->_query)) {
244             throw new ClientException('Query should be a string');
245         }
246 
247         $data = $this->buildData();
248 
249         $tries = 0;
250         while (true) {
251             try {
252                 $response = $this->_connection->post(Urls::URL_CURSOR, $this->getConnection()->json_encode_wrapper($data), array());
253                 return new Cursor($this->_connection, $response->getJson(), $this->getCursorOptions());
254             } catch (ServerException $e) {
255                 if ($tries++ >= $this->_retries) {
256                     throw $e;
257                 }
258                 if ($e->getServerCode() !== 29) {
259                     // 29 is "deadlock detected"
260                     throw $e;
261                 }
262                 // try again
263             }
264         }
265     }

500     /**
501      * Build an array of data to be posted to the server when issuing the statement
502      *
503      * @return array - array of data to be sent to server
504      */
505     private function buildData()
506     {
507         $data = array(
508             self::ENTRY_QUERY => $this->_query,
509             self::ENTRY_COUNT => $this->_doCount,
510             'options' => array(
511                 self::FULL_COUNT => $this->_fullCount
512             )
513         );
514 
515         if ($this->_cache !== null) {
516             $data[Cursor::ENTRY_CACHE] = $this->_cache;
517         }
518 
519         if ($this->_bindVars->getCount() > 0) {
520             $data[self::ENTRY_BINDVARS] = $this->_bindVars->getAll();
521         }
522 
523         if ($this->_batchSize > 0) {
524             $data[self::ENTRY_BATCHSIZE] = $this->_batchSize;
525         }
526         return $data;
527     }
