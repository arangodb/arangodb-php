134     /**
135      * Get a single document (internal method)
136      *
137      * This method is the workhorse for getById() in this handler and the edges handler
138      *
139      * @throws Exception
140      *
141      * @param string $url - the server-side URL being called
142      * @param string $collection - collection id as a string or number
143      * @param mixed $documentId - document identifier
144      * @param array $options - optional, array of options
145      *                            <p>Options are :
146      *                            <li>'_includeInternals' - true to include the internal attributes. Defaults to false</li>
147      *                            <li>'includeInternals' - Deprecated, please use '_includeInternals'.</li>
148      *                            <li>'_ignoreHiddenAttributes' - true to show hidden attributes. Defaults to false</li>
149      *                            <li>'ignoreHiddenAttributes' - Deprecated, please use '_ignoreHiddenAttributes'.</li>
150      *                            <li>'ifMatch' - boolean if given revision should match or not</li>
151      *                            <li>'revision' - The document is returned if it matches/not matches revision.</li>
152      *                            </p>
153      * @internal
154      *
155      * @return array - the document fetched from the server
156      */
157     protected function getDocument($url, $collection, $documentId, array $options = array())
158     {
159         $collection = $this->makeCollection($collection);
160 
161         $url            = UrlHelper::buildUrl($url, array($collection, $documentId));
162         $headerElements = array();
163         if (array_key_exists('ifMatch', $options) && array_key_exists('revision', $options)) {
164             if ($options['ifMatch'] === true) {
165                 $headerElements['If-Match'] = '"' . $options['revision'] . '"';
166             }
167             else {
168                 $headerElements['If-None-Match'] = '"' . $options['revision'] . '"';
169             }
170         }
171 
172         $response = $this->getConnection()->get($url, $headerElements);
173 
174         if ($response->getHttpCode() === 304) {
175             throw new ClientException('Document has not changed.');
176         }
177 
178         return $response->getJson();
179     }

350      *
351      * This is an alias/shortcut to save() and replace(). Instead of having to determine which of the 3 functions to use,
352      * simply pass the document to store() and it will figure out which one to call.
353      *
354      * This will throw if the document cannot be saved or replaced.
355      *
356      * @throws Exception
357      *
358      * @param Document $document - the document to be added, can be passed as a document or an array
359      * @param mixed $collection - collection id as string or number
360      * @param bool|array $options - optional, prior to v1.2.0 this was a boolean value for create. Since v1.2.0 it's an array of options.
361      *                                 <p>Options are :<br>
362      *                                 <li>'create' - create the collection if it does not yet exist.</li>
363      *                                 <li>'waitForSync' -  if set to true, then all removal operations will instantly be synchronised to disk / If this is not specified, then the collection's default sync behavior will be applied.</li>

365      *

367      * @since 1.0
368      */
369     public function store(Document $document, $collection = null, $options = array())
370     {
371         if ($document->getIsNew()) {
372 
373             if ($collection === null) {
374                 throw new ClientException('A collection id is required to store a new document.');
375             }
376 
377             $result = $this->save($collection, $document, $options);
378             $document->setIsNew(false);
379 
380             return $result;
381         }
382         else {
383 
384             if ($collection) {
385                 throw new ClientException('An existing document cannot be stored into a new collection');
386             }
387 
388             return $this->replace($document, $options);
389         }
390     }

393     /**
394      * save a document to a collection
395      *
396      * This will add the document to the collection and return the document's id
397      *
398      * This will throw if the document cannot be saved
399      *
400      * @throws Exception
401      *
402      * @param mixed $collection - collection id as string or number
403      * @param mixed $document - the document to be added, can be passed as a document or an array
404      * @param bool|array $options - optional, prior to v1.2.0 this was a boolean value for create. Since v1.0.0 it's an array of options.
405      *                                 <p>Options are :<br>
406      *                                 <li>'create' - create the collection if it does not yet exist.</li>
407      *                                 <li>'waitForSync' -  if set to true, then all removal operations will instantly be synchronised to disk / If this is not specified, then the collection's default sync behavior will be applied.</li>
408      *                                 </p>
409      *
410      * @return mixed - id of document created
411      * @since 1.0
412      */
413     public function save($collection, $document, $options = array())
414     {
415         $collection = $this->makeCollection($collection);
416 
417         // This preserves compatibility for the old create parameter.
418         $params = $this->validateAndIncludeOldSingleParameterInParams(
419             $options,
420             [],
421             ConnectionOptions::OPTION_CREATE
422         );
423 
424         $params = $this->includeOptionsInParams(
425             $params,
426             $options,
427             array(
428                 'waitForSync' => ConnectionOptions::OPTION_WAIT_SYNC,
429                 'silent' => false
430             )
431         );
432 
433         $this->createCollectionIfOptions($collection, $params);
434 
435         $url = UrlHelper::appendParamsUrl(Urls::URL_DOCUMENT . '/' . $collection, $params);
436 
437         if (is_array($document)) {
438             $data = $document;
439         }
440         else {
441             $data = $document->getAllForInsertUpdate();
442         }
443 
444         $response = $this->getConnection()->post($url, $this->json_encode_wrapper($data));
445         $json     = $response->getJson();
446 
447         if (is_array($document)) {
448             return $json[Document::ENTRY_KEY];
449         }
450 
451         $location = $response->getLocationHeader();
452         if (!$location) {
453             throw new ClientException('Did not find location header in server response');
454         }
455 
456         $id = UrlHelper::getDocumentIdFromLocation($location);
457 
458         $document->setInternalId($json[Document::ENTRY_ID]);
459         $document->setRevision($json[Document::ENTRY_REV]);
460 
461         if ($id !== $document->getId()) {
462             throw new ClientException('Got an invalid response from the server');
463         }
464 
465         $document->setIsNew(false);
466 
467         return $document->getId();
468     }

534     /**
535      * Update an existing document in a collection (internal method)
536      *
537      * @throws Exception
538      *
539      * @param string $url - server-side URL being called
540      * @param string $collection - collection id as string or number
541      * @param mixed $documentId - document id as string or number
542      * @param Document $document - patch document which contains the attributes and values to be updated
543      * @param mixed $options - optional, array of options (see below) or the boolean value for $policy (for compatibility prior to version 1.1 of this method)
544      *                               <p>Options are :
545      *                               <li>'policy' - update policy to be used in case of conflict ('error', 'last' or NULL [use default])</li>
546      *                               <li>'keepNull' - can be used to instruct ArangoDB to delete existing attributes instead setting their values to null. Defaults to true (keep attributes when set to null)</li>
547      *                               <li>'waitForSync' - can be used to force synchronisation of the document update operation to disk even in case that the waitForSync flag had been disabled for the entire collection</li>
548      *                               </p>
549      * @internal
550      *
551      * @return bool - always true, will throw if there is an error
552      */
553     protected function patch($url, $collection, $documentId, Document $document, $options = array())
554     {
555         $collection = $this->makeCollection($collection);
556 
557         // This preserves compatibility for the old policy parameter.
558         $params = $this->validateAndIncludeOldSingleParameterInParams(
559             $options,
560             array(),
561             ConnectionOptions::OPTION_UPDATE_POLICY
562         );
563 
564         $params = $this->includeOptionsInParams(
565             $options,
566             $params,
567             array(
568                 'waitForSync' => $this->getConnectionOption(ConnectionOptions::OPTION_WAIT_SYNC),
569                 'keepNull' => true,
570                 'silent' => false,
571                 'ignoreRevs' => true,
572                 'policy' => ''
573             )
574         );
575 
576 
577         $headers = array();
578         if (isset($params[ConnectionOptions::OPTION_UPDATE_POLICY]) &&
579             $params[ConnectionOptions::OPTION_UPDATE_POLICY] === UpdatePolicy::ERROR
580         ) {
581 
582             $revision = $document->getRevision();
583             if (null !== $revision) {
584                 $params['ignoreRevs'] = false;
585                 $headers['if-match']  = '"' . $revision . '"';
586             }
587         }
588 
589         $url = UrlHelper::buildUrl($url, array($collection, $documentId));
590         $url = UrlHelper::appendParamsUrl($url, $params);
591 
592         $result = $this->getConnection()->patch($url, $this->json_encode_wrapper($document->getAllForInsertUpdate()), $headers);
593         $json   = $result->getJson();
594         $document->setRevision($json[Document::ENTRY_REV]);
595 
596         return true;
597     }
