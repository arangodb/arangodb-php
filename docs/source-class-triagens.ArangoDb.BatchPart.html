 89     public function __construct($batch, $id, $type, $request, $response, $options)

233     /**
234      * Get the batch part identified by the array key (0...n) or its id (if it was set with nextBatchPartId($id) )
235      *
236      * @throws ClientException
237      * @return mixed $partId
238      */
239     public function getProcessedResponse()
240     {
241         $response = $this->getResponse();
242         switch ($this->_type) {
243             case 'getdocument':
244                 $json             = $response->getJson();
245                 $options          = $this->getCursorOptions();
246                 $options['isNew'] = false;
247                 $response         = Document::createFromArray($json, $options);
248                 break;
249             case 'document':
250                 $json = $response->getJson();
251                 if (!isset($json['error']) || $json['error'] === false) {
252                     $id       = $json[Document::ENTRY_ID];
253                     $response = $id;
254                 }
255                 break;
256             case 'getedge':
257                 $json             = $response->getJson();
258                 $options          = $this->getCursorOptions();
259                 $options['isNew'] = false;
260                 $response         = Edge::createFromArray($json, $options);
261                 break;
262             case 'edge':
263                 $json = $response->getJson();
264                 if (!isset($json['error']) || $json['error'] === false) {
265                     $id       = $json[Edge::ENTRY_ID];
266                     $response = $id;
267                 }
268                 break;
269             case 'getcollection':
270                 $json             = $response->getJson();
271                 $options          = $this->getCursorOptions();
272                 $options['isNew'] = false;
273                 $response         = Collection::createFromArray($json, $options);
274                 break;
275             case 'collection':
276                 $json = $response->getJson();
277                 if (!isset($json['error']) || $json['error'] === false) {
278                     $id       = $json[Collection::ENTRY_ID];
279                     $response = $id;
280                 }
281                 break;
282             case 'cursor':
283                 $options          = $this->getCursorOptions();
284                 $options['isNew'] = false;
285                 $response         = new Cursor($this->_batch->getConnection(), $response->getJson(), $options);
286                 break;
287             default:
288                 throw new ClientException('Could not determine response data type.');
289                 break;
290         }
291 
292         return $response;
293     }
