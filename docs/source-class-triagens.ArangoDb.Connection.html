133      * Set an option set for the connection

135      * @throws ClientException

137      * @param string $name - name of option
138      * @param string $value - value of option

140     public function setOption($name, $value)
141     {
142         if ($name === ConnectionOptions::OPTION_ENDPOINT ||
143             $name === ConnectionOptions::OPTION_HOST ||
144             $name === ConnectionOptions::OPTION_PORT ||
145             $name === ConnectionOptions::OPTION_VERIFY_CERT ||
146             $name === ConnectionOptions::OPTION_CIPHERS ||
147             $name === ConnectionOptions::OPTION_ALLOW_SELF_SIGNED
148         ) {
149             throw new ClientException('Must not set option ' . $value . ' after connection is created.');
150         }
151 
152         $this->_options[$name] = $value;
153 
154         // special handling for several options
155         if ($name === ConnectionOptions::OPTION_TIMEOUT) {
156             // set the timeout option: patch the stream of an existing connection
157             if (is_resource($this->_handle)) {
158                 stream_set_timeout($this->_handle, $value);
159             }
160         }
161         else if ($name === ConnectionOptions::OPTION_CONNECTION) {
162             // set keep-alive flag
163             $this->_useKeepAlive = (strtolower($value) === 'keep-alive');
164         }
165         else if ($name === ConnectionOptions::OPTION_DATABASE) {
166             // set database
167             $this->setDatabase($value);
168         }
169 
170         $this->updateHttpHeader();
171     }

306     /**
307      * Recalculate the static HTTP header string used for all HTTP requests in this connection
308      */
309     private function updateHttpHeader()
310     {
311         $this->_httpHeader = HttpHelper::EOL;
312 
313         $endpoint = $this->_options[ConnectionOptions::OPTION_ENDPOINT];
314         if (Endpoint::getType($endpoint) !== Endpoint::TYPE_UNIX) {
315             $this->_httpHeader .= sprintf('Host: %s%s', Endpoint::getHost($endpoint), HttpHelper::EOL);
316         }
317 
318         if (isset($this->_options[ConnectionOptions::OPTION_AUTH_TYPE], $this->_options[ConnectionOptions::OPTION_AUTH_USER])) {
319             // add authorization header
320             $authorizationValue = base64_encode(
321                 $this->_options[ConnectionOptions::OPTION_AUTH_USER] . ':' .
322                 $this->_options[ConnectionOptions::OPTION_AUTH_PASSWD]
323             );
324 
325             $this->_httpHeader .= sprintf(
326                 'Authorization: %s %s%s',
327                 $this->_options[ConnectionOptions::OPTION_AUTH_TYPE],
328                 $authorizationValue,
329                 HttpHelper::EOL
330             );
331         }
332 
333         if (isset($this->_options[ConnectionOptions::OPTION_CONNECTION])) {
334             // add connection header
335             $this->_httpHeader .= sprintf('Connection: %s%s', $this->_options[ConnectionOptions::OPTION_CONNECTION], HttpHelper::EOL);
336         }
337 
338         if ($this->_database === '') {
339             $this->_baseUrl = '/_db/_system';
340         }
341         else {
342             $this->_baseUrl = '/_db/' . urlencode($this->_database);
343         }
344     }

346     /**
347      * Get a connection handle
348      *
349      * If keep-alive connections are used, the handle will be stored and re-used
350      *
351      * @throws ClientException
352      * @return resource - connection handle
353      */
354     private function getHandle()
355     {
356         if ($this->_useKeepAlive && $this->_handle && is_resource($this->_handle)) {
357             // keep-alive and handle was created already
358             $handle = $this->_handle;
359 
360             // check if connection is still valid
361             if (!feof($handle)) {
362                 // connection still valid
363                 return $handle;
364             }
365 
366             // close handle
367             @fclose($this->_handle);
368             $this->_handle = 0;
369 
370             if (!$this->_options[ConnectionOptions::OPTION_RECONNECT]) {
371                 // if reconnect option not set, this is the end
372                 throw new ClientException('Server has closed the connection already.');
373             }
374         }
375 
376         // no keep-alive or no handle available yet or a reconnect
377         $handle = HttpHelper::createConnection($this->_options);
378 
379         if ($this->_useKeepAlive && is_resource($handle)) {
380             $this->_handle = $handle;
381         }
382 
383         return $handle;
384     }

386     /**
387      * Execute an HTTP request and return the results
388      *
389      * This function will throw if no connection to the server can be established or if
390      * there is a problem during data exchange with the server.
391      *
392      * will restore it.
393      *
394      * @throws Exception
395      *
396      * @param string $method - HTTP request method
397      * @param string $url - HTTP URL
398      * @param string $data - data to post in body
399      * @param array $customHeaders - any array containing header elements
400      *
401      * @return HttpResponse
402      */
403     private function executeRequest($method, $url, $data, array $customHeaders = array())
404     {
405         assert($this->_httpHeader !== '');
406         $wasAsync = false;
407         if (is_array($customHeaders) && isset($customHeaders[HttpHelper::ASYNC_HEADER])) {
408             $wasAsync = true;
409         }
410 
411         HttpHelper::validateMethod($method);
412         $url = $this->_baseUrl . $url;
413 
414         // create request data
415         if ($this->_batchRequest === false) {
416 
417             if ($this->_captureBatch === true) {
418                 $this->_options->offsetSet(ConnectionOptions::OPTION_BATCHPART, true);
419                 $request = HttpHelper::buildRequest($this->_options, $this->_httpHeader, $method, $url, $data, $customHeaders);
420                 $this->_options->offsetSet(ConnectionOptions::OPTION_BATCHPART, false);
421             }
422             else {
423                 $request = HttpHelper::buildRequest($this->_options, $this->_httpHeader, $method, $url, $data, $customHeaders);
424             }
425 
426             if ($this->_captureBatch === true) {
427                 $batchPart = $this->doBatch($method, $request);
428                 if (null !== $batchPart) {
429                     return $batchPart;
430                 }
431             }
432         }
433         else {
434             $this->_batchRequest = false;
435 
436             $this->_options->offsetSet(ConnectionOptions::OPTION_BATCH, true);
437 
438             $request = HttpHelper::buildRequest($this->_options, $this->_httpHeader, $method, $url, $data, $customHeaders);
439             $this->_options->offsetSet(ConnectionOptions::OPTION_BATCH, false);
440         }
441 
442         $request = HttpHelper::buildRequest($this->_options, $this->_httpHeader, $method, $url, $data, $customHeaders);
443 
444         $traceFunc = $this->_options[ConnectionOptions::OPTION_TRACE];
445         if ($traceFunc) {
446             // call tracer func
447             if ($this->_options[ConnectionOptions::OPTION_ENHANCED_TRACE]) {
448                 list($header) = HttpHelper::parseHttpMessage($request, $url, $method);
449                 $headers = HttpHelper::parseHeaders($header);
450                 $traceFunc(new TraceRequest($headers[2], $method, $url, $data));
451             }
452             else {
453                 $traceFunc('send', $request);
454             }
455         }
456 
457 
458         // open the socket. note: this might throw if the connection cannot be established
459         $handle = $this->getHandle();
460 
461         if ($handle) {
462             // send data and get response back
463 
464             if ($traceFunc) {
465                 // only issue syscall if we need it
466                 $startTime = microtime(true);
467             }
468 
469             $result = HttpHelper::transfer($handle, $request);
470 
471             if ($traceFunc) {
472                 // only issue syscall if we need it
473                 $timeTaken = microtime(true) - $startTime;
474             }
475 
476             $status = socket_get_status($handle);
477             if ($status['timed_out']) {
478                 throw new ClientException('Got a timeout while waiting for the server\'s response', 408);
479             }
480 
481             if (!$this->_useKeepAlive) {
482                 // must close the connection
483                 fclose($handle);
484             }
485 
486             $response = new HttpResponse($result, $url, $method, $wasAsync);
487 
488             if ($traceFunc) {
489                 // call tracer func
490                 if ($this->_options[ConnectionOptions::OPTION_ENHANCED_TRACE]) {
491                     $traceFunc(
492                         new TraceResponse(
493                             $response->getHeaders(), $response->getHttpCode(), $response->getBody(),
494                             $timeTaken
495                         )
496                     );
497                 }
498                 else {
499                     $traceFunc('receive', $result);
500                 }
501             }
502 
503             return $response;
504         }
505 
506         throw new ClientException('Whoops, this should never happen');
507     }

509     /**
510      * Parse the response return the body values as an assoc array
511      *
512      * @throws Exception
513      *
514      * @param HttpResponse $response - the response as supplied by the server
515      *
516      * @return HttpResponse
517      */
518     public function parseResponse(HttpResponse $response)
519     {
520         $httpCode = $response->getHttpCode();
521 
522         if ($httpCode < 200 || $httpCode >= 400) {
523             // failure on server
524 
525             $body = $response->getBody();
526             if ($body !== '') {
527                 // check if we can find details in the response body
528                 $details = json_decode($body, true);
529                 if (is_array($details) && isset($details['errorMessage'])) {
530                     // yes, we got details
531                     $exception = new ServerException($details['errorMessage'], $details['code']);
532                     $exception->setDetails($details);
533                     throw $exception;
534                 }
535             }
536 
537             // no details found, throw normal exception
538             throw new ServerException($response->getResult(), $httpCode);
539         }
540 
541         return $response;
542     }

675     /**
676      * This function checks that the encoding of the keys and
677      * values of the array are utf-8, recursively.
678      * It will raise an exception if it encounters wrong encoded strings.
679      *
680      * @param array $data the data to check
681      *
682      * @throws ClientException
683      */
684     public static function check_encoding($data)
685     {
686         if (!is_array($data)) {
687             return;
688         }
689 
690         foreach ($data as $key => $value) {
691             if (!is_array($value)) {
692                 // check if the multibyte library function is installed and use it.
693                 if (function_exists('mb_detect_encoding')) {
694                     // check with mb library
695                     if (is_string($key) && mb_detect_encoding($key, 'UTF-8', true) === false) {
696                         throw new ClientException('Only UTF-8 encoded keys allowed. Wrong encoding in key string: ' . $key);
697                     }
698                     if (is_string($value) && mb_detect_encoding($value, 'UTF-8', true) === false) {
699                         throw new ClientException('Only UTF-8 encoded values allowed. Wrong encoding in value string: ' . $value);
700                     }
701                 }
702                 else {
703                     // fallback to preg_match checking
704                     if (is_string($key) && self::detect_utf($key) == false) {
705                         throw new ClientException('Only UTF-8 encoded keys allowed. Wrong encoding in key string: ' . $key);
706                     }
707                     if (is_string($value) && self::detect_utf($value) == false) {
708                         throw new ClientException('Only UTF-8 encoded values allowed. Wrong encoding in value string: ' . $value);
709                     }
710                 }
711             }
712             else {
713                 self::check_encoding($value);
714             }
715         }
716     }

720      * This is a json_encode() wrapper that also checks if the data is utf-8 conform.
721      * internally it calls the check_encoding() method. If that method does not throw
722      * an Exception, this method will happily return the json_encoded data.
